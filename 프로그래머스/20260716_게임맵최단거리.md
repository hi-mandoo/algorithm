## 게임 맵 최단거리

- [문제 링크](https://school.programmers.co.kr/learn/courses/30/lessons/1844)

## 풀이

다익스트라

## 코드

```java
import java.util.Comparator;
import java.util.PriorityQueue;
import java.util.Queue;

class Solution {
    public int solution(int[][] maps) {
        Queue<Character> q = new PriorityQueue<>(Comparator.comparingInt(character -> character.distance));
        q.add(new Character(0, 0, 1));

        int[] directionX = {0, 0, -1, 1};
        int[] directionY = {-1, 1, 0, 0};

        boolean[][] visited = new boolean[maps.length][maps[0].length];
        visited[0][0] = true;


        while (!q.isEmpty()) {
            Character curr = q.poll();

            if (curr.x == maps.length - 1 && curr.y == maps[0].length - 1) {
                return curr.distance;
            }

            for (int i = 0; i < 4; i++) {
                int x = curr.x + directionX[i];
                int y = curr.y + directionY[i];

                if (x < 0 || y < 0 || x >= maps.length || y >= maps[0].length || maps[x][y] == 0 || visited[x][y])
                    continue;

                visited[x][y] = true;
                q.add(new Character(x, y, curr.distance + 1));
            }
        }

        return -1;
    }

    public static class Character {
        int x;
        int y;
        int distance;

        public Character(int x, int y, int distance) {
            this.x = x;
            this.y = y;
            this.distance = distance;
        }
    }
}
```
