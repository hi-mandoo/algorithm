## 카카오프렌즈 컬러링북

- [문제 링크](https://school.programmers.co.kr/learn/courses/30/lessons/1829)

### 풀이

BFS로 풀이했다.  
다만 색상이나 최대 크기를 체크해줘야하는 부분이 있어 주의가 필요하다.

### 코드

```java
import java.util.ArrayDeque;
import java.util.Queue;

class Solution {

    int[][] picture;
    boolean[][] isVisited;
    int numberOfArea = 0;
    int maxSizeOfOneArea = 0;

    int[] dx = {0, 0, -1, 1};
    int[] dy = {-1, 1, 0, 0};

    public int[] solution(int m, int n, int[][] picture) {
        this.picture = picture;
        isVisited = new boolean[m][n];

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (isVisited[i][j] || picture[i][j] == 0) continue;
                bfs(i, j);
            }
        }

        int[] answer = new int[2];
        answer[0] = numberOfArea;
        answer[1] = maxSizeOfOneArea;
        return answer;
    }

    private void bfs(int x, int y) {
        numberOfArea++;
        int size = 1;
        int color = picture[x][y];

        Queue<Node> queue = new ArrayDeque<>();
        queue.add(new Node(x, y));
        isVisited[x][y] = true;

        while (!queue.isEmpty()) {
            Node curr = queue.poll();

            for (int i = 0; i < 4; i++) {
                int currx = curr.x + dx[i];
                int curry = curr.y + dy[i];

                if (currx < 0 || currx >= picture.length || curry < 0 || curry >= picture[0].length || isVisited[currx][curry] || picture[currx][curry] != color)
                    continue;

                isVisited[currx][curry] = true;
                size++;
                queue.add(new Node(currx, curry));
            }
        }
        maxSizeOfOneArea = Math.max(maxSizeOfOneArea, size);
    }

    static class Node {
        int x;
        int y;

        public Node(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }
}
```
