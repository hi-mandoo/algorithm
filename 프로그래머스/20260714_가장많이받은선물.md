## 가장 많이 받은 선물

- [문제 링크](https://school.programmers.co.kr/learn/courses/30/lessons/258712)

## 코드

```java
import java.util.HashMap;
import java.util.StringTokenizer;

class Solution {
    public int solution(String[] friends, String[] gifts) {
        // index 계산
        HashMap<String, Integer> map = new HashMap<>();
        for (int i = 0; i < friends.length; i++) {
            map.put(friends[i], i);
        }

        int[][] graph = new int[friends.length][friends.length];

        for (int i = 0; i < gifts.length; i++) {
            StringTokenizer st = new StringTokenizer(gifts[i]);
            graph[map.get(st.nextToken())][map.get(st.nextToken())] += 1;
        }

        // 선물 지수 계산
        int[] giftScore = new int[friends.length];

        for (int i = 0; i < friends.length; i++) {
            for (int j = 0; j < friends.length; j++) {
                if (i == j) continue;
                giftScore[i] += graph[i][j];
                giftScore[i] -= graph[j][i];
            }
        }

        int[] receivedGifts = new int[gifts.length];

        for (int i = 0; i < friends.length; i++) {
            for (int j = i + 1; j < friends.length; j++) {
                if (i == j) continue;

                if (graph[i][j] > graph[j][i]) receivedGifts[i]++;
                else if (graph[i][j] < graph[j][i]) receivedGifts[j]++;
                else if (graph[i][j] == graph[j][i]) {
                    if (giftScore[i] > giftScore[j]) receivedGifts[i]++;
                    else if (giftScore[i] < giftScore[j]) receivedGifts[j]++;
                }
            }
        }

        int answer = 0;
        for (int i = 0; i < friends.length; i++) {
            answer = Math.max(answer, receivedGifts[i]);
        }

        return answer;
    }
}
```
